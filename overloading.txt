//Wap to find volume of cube, cuboid and cylinder using the concept of function overloading.
 #include<iostream>
 using namespace std;
 void volume(float l)
 {
 cout<<"the volume of cube is:"<<l*l*l<<endl;

 }
void volume(float l, float b, float h)
{
    cout<<"the volume of cuboid is:"<<l*b*h<<endl;

}
void volume(float r , float h)
{
    cout<<"the volume of cylinder is:"<<3.14*r*h<<endl;

}
int main(){
    volume(3.5);
    volume(3.5 , 6.5 , 3.5);
    volume(3.5 , 5.5);
    return 0;
    
}
